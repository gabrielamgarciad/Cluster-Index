
getvoxeldims = function(folder, ...){



fname = oro.dicom::readDICOM(folder, verbose = TRUE)
  
 pixelLength = as.numeric(strsplit(fname$hdr[[1]]$value[fname$hdr[[1]]$name %in% 
"PixelSpacing"], " ")[[1]][1])
    thickness = unique(oro.dicom::extractHeader(fname$hdr, "SliceThickness"))
    returninfo = c(pixelLength, pixelLength, thickness)
    returninfo
}
